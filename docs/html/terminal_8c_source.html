1032 {

1034  char pressed, status;
1035  // to catch arrrow keypressed we need to read 3 characters
1036  // 2 delimiters and 1 for the actual button
1037  // implemented as a simple FSM
1038  // the 8 lsb represent (from left to right)
1039  // BACKSPACE SPACEBAR ENTER TAB RIGHT LEFT DOWN UP pressed.
1040  pressed = 0;
1041  status = 0;
1042  while ((key = poll_keypress()) > 0)
1043  {
1044  if (key == 27)
1045  {
1046  // first step of the FSM
1047  // first escape character found
1048  status |= 0b00000001;
1049  }
1050  else if (key == 91 && status & 0b00000001)
1051  {
1052  // second step of the FSM
1053  // second escape character found
1054  status |= 0b00000010;
1055  }
1056  else if (key >= 65 && key <= 68 && status & 0b00000010)
1057  {
1058  // last step of the FSM
1059  // key character found
1060  switch (key)
1061  {
1062  case 65:
1063  pressed |= 0b00000001; // up
1064  break;
1065  case 66:
1066  pressed |= 0b00000010; // down
1067  break;
1068  case 67:
1069  pressed |= 0b00000100; // left
1070  break;
1071  case 68:
1072  pressed |= 0b00001000; // right
1073  break;
1074 
1075  default:
1076  break;
1077  }
1078 
1079  pressed &= 0b00001111;
1080  status &= 0b11110000;
1081  }
1082  else
1083  {
1084  // some keys are not escaped
1085  status = 0;
1086  switch (key)
1087  {
1088  case 9:
1089  pressed |= 0b00010000; // tab
1090  break;
1091  case 10:
1092  pressed |= 0b00100000; // enter
1093  break;
1094  case 32:
1095  pressed |= 0b01000000; // space bar
1096  break;
1097  case 127:
1098  pressed |= 0b10000000; // backspace
1099  break;
1100  default:

1102  }
